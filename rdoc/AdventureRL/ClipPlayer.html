def initialize settings = {}
  @settings = DEFAULT_SETTINGS.merge settings
  set_mask_from @settings.get(:mask)
  @clips = {}
  load_clips *@settings.get(:clips)  if (@settings.get(:clips))
  @deltatime      = Deltatime.new

  @active_clip    = nil
  @active_image   = nil
  @playing        = false
  @accumulated_dt = 0.0

def load_clip clip
  error(
    "Cannot load clip `#{clip.to_s}:#{clip.class.name}'",
    "Argument must be a `AdventureRL::Clip'"
  )  unless (clip.is_a? Clip)
  @clips[clip.get_name] = clip
end

    play(clipname) click to toggle source

    Start playing loaded Clip with the name clipname. clipname can either be the name of a previously loaded _(see #load_clip)_ Clip, or an instance of AdventureRL::Clip.

    # File lib/AdventureRL/ClipPlayer.rb, line 77
def play clipname
  if (clipname.is_a? Clip)
    load_clip clipname
    set_active_clip clipname
  else
    clip = get_clip clipname
    set_active_clip clip
  end
  @playing = true
end
