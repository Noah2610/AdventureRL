def initialize settings = {}
  @settings = DEFAULT_SETTINGS.merge settings
  set_mask_from @settings.get(:mask)
  @clips = {}
  load_clips *@settings.get(:clips)  if (@settings.get(:clips))
  @speed                    = @settings.get(:speed)
  @deltatime                = Deltatime.new
  @image_index              = 0
  @active_clip              = nil
  @active_image             = nil
  @playing                  = false
  @current_time             = 0.0
  @last_current_time        = 0.0
  @accumulated_current_time = 0.0
end

    Draw the current image in the currently active Clip. Switch to the next image automatically when enough time passes.

    # File lib/AdventureRL/ClipPlayer.rb, line 156
def draw
  draw_current_image
  return  unless (is_playing?)
  set_image_index
  @current_time             += @deltatime.dt * @speed
  @accumulated_current_time += @current_time - @last_current_time
  @last_current_time         = @current_time
  @deltatime.update
end

def load_clip clip
  error(
    "Cannot load clip `#{clip.to_s}:#{clip.class.name}'",
    "Argument must be a `AdventureRL::Clip'"
  )  unless (clip.is_a? Clip)
  @clips[clip.get_name] = clip
end

    Start playing loaded Clip with the name clipname. clipname can either be the name of a previously loaded (see load_clip) Clip, or an instance of Clip.

    # File lib/AdventureRL/ClipPlayer.rb, line 108
def play clipname
  if (clipname.is_a? Clip)
    load_clip clipname
    set_active_clip clipname
  else
    clip = get_clip clipname
    set_active_clip clip
  end
  @playing = true
  @deltatime.reset
end
